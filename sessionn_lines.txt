//@version=6
indicator("Session Lines", overlay = true, max_boxes_count = 500, max_lines_count = 500, max_labels_count = 500)

// Inputs (existing)
// =======================
bool showSessionNames     = input.bool(true, "Show session names")
bool showSessionOC        = input.bool(true, "Draw session open and close lines")
bool showSessionTickRange = input.bool(true, "Show tick range for each session")
bool showSessionAverage   = input.bool(true, "Show average price per session")

const string TZ_TOOLTIP_TEXT = "The session's time zone, specified in either GMT notation (e.g., 'GMT-5') or as an IANA time zone database name (e.g., 'America/New_York')." +
 " We recommend the latter since it includes other time-related changes, such as daylight savings."

// =======================
// Session High/Low extension lines
// =======================
grpHL = "Session High/Low Lines"
showSessionHL      = input.bool(true, "Draw session high/low lines", group=grpHL)
extendDays         = input.int(3, "Extend right (days)", minval=1, maxval=365, group=grpHL)
hlLineWidth        = input.int(1, "Line width", minval=1, maxval=5, group=grpHL)
hlLineStyleStr     = input.string("Solid", "Line style", options=["Solid","Dashed","Dotted"], group=grpHL)

// =======================
// Labels for session high/low
// =======================
grpHLLabels        = "Session High/Low Labels"
showHLLabels       = input.bool(true, "Show High/Low labels", group=grpHLLabels)
labelOnFinalize    = input.bool(true, "Place labels when session ends (locked)", group=grpHLLabels)
labelTextSizeStr   = input.string("Small", "Label text size", options=["Tiny","Small","Normal","Large","Huge"], group=grpHLLabels)
labelOffsetBars    = input.int(0, "Label X offset (bars)", minval=-50, maxval=200, group=grpHLLabels)
// FIX: step must be const; use a constant like 0.25 (you can change it in UI via typing anyway)
labelOffsetPrice   = input.float(0.0, "Label Y offset (price)", step=0.25, group=grpHLLabels)

lineStyleFromStr(string s) =>
    s == "Dashed" ? line.style_dashed : s == "Dotted" ? line.style_dotted : line.style_solid

labelSizeFromStr(string s) =>
    s == "Tiny" ? size.tiny :
     s == "Small" ? size.small :
     s == "Normal" ? size.normal :
     s == "Large" ? size.large : size.huge

hlLineStyle  = lineStyleFromStr(hlLineStyleStr)
lblTextSize  = labelSizeFromStr(labelTextSizeStr)

int dayMs    = 24 * 60 * 60 * 1000
int extendMs = extendDays * dayMs

// =======================
// Sessions (existing)
// =======================
const string FIRST_SESSION_GROUP = "First Session"
showFirst         = input.bool(true, "Show session", group = FIRST_SESSION_GROUP, display = display.none)
firstSessionName  = input.string("Tokyo", "Displayed name", group = FIRST_SESSION_GROUP, display = display.none, active = showFirst)
firstSessionTime  = input.session("1000-1500", "Session time", group = FIRST_SESSION_GROUP, display = display.none, active = showFirst)
firstSessionTZ    = input.string("Asia/Tokyo", "Session timezone", group = FIRST_SESSION_GROUP, display = display.none, tooltip = TZ_TOOLTIP_TEXT, active = showFirst)
firstSessionColor = input.color(color.new(#2962FF, 85), "Session color", group = FIRST_SESSION_GROUP, active = showFirst)

showSecond         = input.bool(true, "Show session", group = SECOND_SESSION_GROUP, display = display.none)
secondSessionName  = input.string("London", "Displayed name", group = SECOND_SESSION_GROUP, display = display.none, active = showSecond)
secondSessionTime  = input.session("0830-1415", "Session time", group = SECOND_SESSION_GROUP, display = display.none, active = showSecond)
secondSessionTZ    = input.string("Europe/London", "Session timezone", group = SECOND_SESSION_GROUP, display = display.none, tooltip = TZ_TOOLTIP_TEXT, active = showSecond)
secondSessionColor = input.color(color.new(#FF9800, 85), "Session color", group = SECOND_SESSION_GROUP, active = showSecond)

const string THIRD_SESSION_GROUP = "Third session"
showThird         = input.bool(true, "Show session", group = THIRD_SESSION_GROUP, display = display.none)
thirdSessionName  = input.string("New York", "Displayed name", group = THIRD_SESSION_GROUP, display = display.none, active = showThird)
thirdSessionTime  = input.session("0930-1600", "Session time", group = THIRD_SESSION_GROUP, display = display.none, active = showThird)
thirdSessionTZ    = input.string("America/New_York", "Session timezone", group = THIRD_SESSION_GROUP, display = display.none, tooltip = TZ_TOOLTIP_TEXT, active = showThird)
thirdSessionColor = input.color(color.new(#089981, 85), "Session color", group = THIRD_SESSION_GROUP, active = showThird)

// =======================
// Types
// =======================
type SessionDisplay
    box   sessionBox
    label sessionLabel
    line  openLine
    line  avgLine
    line  closeLine
    float sumClose
    int   numOfBars
    line  hiLine
    line  loLine
    float sessHigh
    float sessLow
    bool  finalized
    label hiLabel
    label loLabel

type SessionInfo
    color  color
    string name
    string session
    string timezone
    SessionDisplay active = na

// =======================
// Methods
// =======================
method setName(SessionDisplay this, string name) =>
    sessionLabel = this.sessionLabel
    sessionBox = this.sessionBox
    boxText = array.new<string>()
    if showSessionTickRange
        boxText.push("Range: " + str.tostring((sessionBox.get_top() - sessionBox.get_bottom()) / syminfo.mintick, format.mintick))
    if showSessionAverage
        boxText.push("Avg: " + str.tostring(this.sumClose / this.numOfBars, format.mintick))
    if showSessionNames
        boxText.push(name)

    sessionLabel.set_y(sessionBox.get_bottom())
    sessionLabel.set_text(array.join(boxText, "\n"))

// Always return SessionDisplay (avoids series label/void return inference issues)
method createOrUpdateHLLabels(SessionInfo info, SessionDisplay sd, bool finalPlacement) =>
    bool should = showHLLabels and showSessionHL and (not labelOnFinalize or finalPlacement)
    if should
        int x = bar_index + labelOffsetBars

        float hiY = sd.sessHigh + labelOffsetPrice
        float loY = sd.sessLow  - labelOffsetPrice

        color txtCol = color.new(info.color, 0)
        color bgCol  = color.new(info.color, 90)

        string hiTxt = info.name + " High"
        string loTxt = info.name + " Low"

        if na(sd.hiLabel)
            sd.hiLabel := label.new(x, hiY, hiTxt, style=label.style_label_left, textcolor=txtCol, color=bgCol, size=lblTextSize)
        else
            label.set_x(sd.hiLabel, x)
            label.set_y(sd.hiLabel, hiY)
            label.set_text(sd.hiLabel, hiTxt)

        if na(sd.loLabel)
            sd.loLabel := label.new(x, loY, loTxt, style=label.style_label_left, textcolor=txtCol, color=bgCol, size=lblTextSize)
        else
            label.set_x(sd.loLabel, x)
            label.set_y(sd.loLabel, loY)
            label.set_text(sd.loLabel, loTxt)

    sd

method createSessionDisplay(SessionInfo this) =>
    boxColor = this.color
    opaqueColor = color.new(boxColor, 0)

    float initHi = high
    float initLo = low

    line _hi = na
    line _lo = na
    if showSessionHL
        _hi := line.new(time, initHi, time, initHi, xloc=xloc.bar_time, color=opaqueColor, style=hlLineStyle, width=hlLineWidth)
        _lo := line.new(time, initLo, time, initLo, xloc=xloc.bar_time, color=opaqueColor, style=hlLineStyle, width=hlLineWidth)

    dis = SessionDisplay.new(
      sessionBox = box.new(bar_index, high, bar_index, low, bgcolor = boxColor, border_color = na),
      sessionLabel = label.new(bar_index, low, "", style = label.style_label_upper_left, textalign = text.align_left, textcolor = opaqueColor, color = color(na)),
      openLine   = showSessionOC ? line.new(bar_index, open, bar_index, open, color = opaqueColor, style = line.style_dashed, width = 1) : na,
      closeLine  = showSessionOC ? line.new(bar_index, close, bar_index, close, color = opaqueColor, style = line.style_dashed, width = 1) : na,
      avgLine    = showSessionAverage ? line.new(bar_index, close, bar_index, close, style = line.style_dotted, width = 2, color = opaqueColor) : na,
      sumClose   = close,
      numOfBars  = 1,
      hiLine     = _hi,
      loLine     = _lo,
      sessHigh   = initHi,
      sessLow    = initLo,
      finalized  = false,
      hiLabel    = na,
      loLabel    = na
    )

    linefill.new(dis.openLine, dis.closeLine, boxColor)
    dis.setName(this.name)

    dis := this.createOrUpdateHLLabels(dis, false)
    this.active := dis

method updateSessionDisplay(SessionInfo this) =>
    SessionDisplay sd = this.active
    box   sessionBox  = sd.sessionBox
    line  openLine    = sd.openLine
    line  closeLine   = sd.closeLine
    line  avgLine     = sd.avgLine

    sd.sumClose += close
    sd.numOfBars += 1

    sessionBox.set_top(math.max(sessionBox.get_top(), high))
    sessionBox.set_bottom(math.min(sessionBox.get_bottom(), low))
    sessionBox.set_right(bar_index)

    sd.sessHigh := math.max(sd.sessHigh, high)
    sd.sessLow  := math.min(sd.sessLow, low)

    if showSessionHL
        if not na(sd.hiLine)
            line.set_x2(sd.hiLine, time)
            line.set_y1(sd.hiLine, sd.sessHigh)
            line.set_y2(sd.hiLine, sd.sessHigh)
        if not na(sd.loLine)
            line.set_x2(sd.loLine, time)
            line.set_y1(sd.loLine, sd.sessLow)
            line.set_y2(sd.loLine, sd.sessLow)

    sd := this.createOrUpdateHLLabels(sd, false)
    sd.setName(this.name)

    if showSessionOC
        line.set_x2(openLine, bar_index)
        line.set_x2(closeLine, bar_index)
        line.set_y1(closeLine, close)
        line.set_y2(closeLine, close)

    if showSessionAverage
        line.set_x2(avgLine, bar_index)
        float avg = sd.sumClose / sd.numOfBars
        line.set_y1(avgLine, avg)
        line.set_y2(avgLine, avg)

    this.active := sd

method finalizeHL(SessionInfo this) =>
    if showSessionHL
        SessionDisplay sd = this.active
        bool ok = not na(sd) and not sd.finalized
        if ok
            int endT   = time
            int rightT = endT + extendMs

            if not na(sd.hiLine)
                line.set_x1(sd.hiLine, endT)
                line.set_x2(sd.hiLine, rightT)
                line.set_y1(sd.hiLine, sd.sessHigh)
                line.set_y2(sd.hiLine, sd.sessHigh)

            if not na(sd.loLine)
                line.set_x1(sd.loLine, endT)
                line.set_x2(sd.loLine, rightT)
                line.set_y1(sd.loLine, sd.sessLow)
                line.set_y2(sd.loLine, sd.sessLow)

            sd := this.createOrUpdateHLLabels(sd, true)

            sd.finalized := true
            this.active := sd

method update(SessionInfo this) =>
    bool isChange  = timeframe.change("1D")
    bool inSession = not na(time("", this.session, this.timezone))

    if inSession
        if na(this.active) or isChange
            this.createSessionDisplay()
        else
            this.updateSessionDisplay()
    else
        if not na(this.active)
            this.finalizeHL()
            this.active := na

getSessionInfos() =>
    array<SessionInfo> sessionInfos = array.new<SessionInfo>()
    if showFirst
        sessionInfos.push(SessionInfo.new(firstSessionColor, firstSessionName, firstSessionTime, firstSessionTZ))
    if showSecond
        sessionInfos.push(SessionInfo.new(secondSessionColor, secondSessionName, secondSessionTime, secondSessionTZ))
    if showThird
        sessionInfos.push(SessionInfo.new(thirdSessionColor, thirdSessionName, thirdSessionTime, thirdSessionTZ))
    sessionInfos

var array<SessionInfo> sessionInfos = getSessionInfos()

if timeframe.isdwm
    runtime.error("This indicator can only be used on intraday timeframes.")

for info in sessionInfos
    info.update()
